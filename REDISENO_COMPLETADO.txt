┌─────────────────────────────────────────────────────────────────────────────┐
│ 🎓 CÓMO USAR EL NUEVO TEMA                                                  │
└─────────────────────────────────────────────────────────────────────────────┘

1. IMPORTAR:
   import 'package:dni_connect/core/theme/app_theme.dart';

2. DETECTAR DARK MODE:
   final isDarkMode = Theme.of(context).brightness == Brightness.dark;

3. USAR COLORES:
   color: isDarkMode ? AppTheme.colorDark : AppTheme.colorLight;

4. APLICAR BORDER RADIUS:
   borderRadius: BorderRadius.circular(24);

5. AGREGAR SOMBRA:
   boxShadow: [BoxShadow(color: Colors.black.withValues(alpha: 0.1), ...)];

6. USAR TIPOGRAFÍA:
   style: GoogleFonts.poppins(fontSize: 18, fontWeight: FontWeight.w700);
